    |H|\sin{35}\textbf{i} +|H|\cos{35}\textbf{j} - |G|\sin{45}\textbf{i} +|G|\cos{45}\textbf{j} + 40\textbf{j} = 0 \\
    \implies |H|\sin{35} - |G|\sin{45} = 0 \quad \text{and} \quad |H|\cos{35} + |G|\cos{45} + 40 = 0 \\
    \implies |H| = \frac{|G|\sin{45}}{\sin{35}} \quad \text{and} \quad |G| = \frac{-40 - |H|\cos{35}}{\cos{45}}\\
\end{align*}
