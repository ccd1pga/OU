\begin{align*}
\stext{Given \( x=3 \) and \( y=-13 \)}\\[8pt]
    -2x -y + \sqrt{y^2 + 5x}&=\\[8pt]
\stext{Substitute \( x=3 \) and \( y=-13 \)}\\[8pt]
    &= -2\rb{3} -\rb{-13} + \sqrt{\rb{-13}^{2} + 5\rb{3}}\\[8pt]
\stext{Simplify}\\[8pt]
    &= -6 + 13 + \sqrt{169 + 15}\\[8pt]
    &= 7 + \sqrt{184}\\[8pt]
    &= 7 + \sqrt4 \times \sqrt{46}\\[8pt]
    &= 7 + 2\sqrt{46}
\end{align*}
