\begin{question}
\[ 1+3+5+\cdots+(2n-1) = n^2 \]

\begin{align*}
\stext{Prove for \(n=4\)}
1+3+5+(2(4)-1) &= 1+3+5+7 \\[8pt]
&= 16 \\[8pt]
&= 4^{2} \\[8pt]
\end{align*}
\end{question}
